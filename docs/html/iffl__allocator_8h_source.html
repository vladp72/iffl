21 //
22 // This class can be used with Polimorfic Memory Allocator.
23 // Forward Linked List has typedef for PMR called pmr_flat_forward_list.
24 //
25 // Sample usage:
26 // iffl::debug_memory_resource dbg_memory_resource;
27 // iffl::pmr_flat_forward_list<FLAT_FORWARD_LIST_TEST> ffl{ &dbg_memory_resource };
28 // Code should guarantee that memory resource is destroyed after all containers that
29 // use it are destroyed.
30 //
31 // This calss provides some basic buffer overrun and underrun detection, and
32 // also diring destruction checks that we did not leak any allocated blocks.
33 //
34 // |---------------------|-------------------|----------------------------------|------------|----------------------------------|
35 // sizeof(size_t) bytes pointer to sizeof(size_t) prefix user data sizeof(size_t) suffix
36 // offset to the end of memory resource 0xBEEFABCD - allocated block filled with 0xDEADABCD - allocated block
37 // allocation 0xBEEFABCE - deallocated block 0xFE 0xDEADABCE - deallocated block
38 //
39 // On deallocation it checks that prefix and suffix contain expected pattern
40 // On allocation user buffer is filled with 0xFE to help detecting use of uninitialized memory
41 //
42 class debug_memory_resource
43  : public std::pmr::memory_resource {

63  using pattern_type = size_t;
64 
65  struct prefix_t {
66  size_t size;
67  size_t alignment;

69  pattern_type pattern;
70  };
71

73  pattern_type pattern;
74  };

76  constexpr static size_t min_allocation_size{ sizeof(prefix_t) + sizeof(suffix_t) };
77 
78  constexpr static pattern_type busy_block_prefix_pattern{ 0xBEEFABCD };
79  constexpr static pattern_type free_block_prefix_pattern{ 0xBEEFABCE };
80  constexpr static pattern_type busy_block_suffix_pattern{ 0xDEADABCD };
81  constexpr static pattern_type free_block_suffix_pattern{ 0xDEADABCE };
82 
83  constexpr static int fill_pattern{ 0xFE };
84 
85  void* do_allocate(size_t bytes, size_t alignment) override {
86 
87  size_t bytes_with_debug_info = bytes + min_allocation_size;
88  void *ptr = _aligned_malloc(bytes_with_debug_info, alignment);
89  if (nullptr == ptr) {
90  throw std::bad_alloc{};
91  }
92 
93  prefix_t *prefix{ reinterpret_cast<prefix_t *>(ptr) };
94  suffix_t *suffix{ reinterpret_cast<suffix_t *>( reinterpret_cast<char *>(ptr) + bytes_with_debug_info) - 1 };
95 
96  prefix->size = bytes_with_debug_info;
97  prefix->alignment = alignment;
98  prefix->memory_resource = this;
99  prefix->pattern = busy_block_prefix_pattern;
100  suffix->pattern = busy_block_suffix_pattern;
101 
102  increment_busy_block_count();
103 
104  char *user_buffer = reinterpret_cast<char *>(prefix + 1);
105  fill_buffer(user_buffer, fill_pattern, bytes);
106 
107  return user_buffer;
108  }
109 
110  void do_deallocate(void* p, size_t bytes, size_t alignment) noexcept override {
111 
112  decrement_busy_block_count();
113 
114  prefix_t *prefix{ reinterpret_cast<prefix_t *>(p) - 1 };
115  FFL_CODDING_ERROR_IF_NOT(prefix->pattern == busy_block_prefix_pattern);
116  FFL_CODDING_ERROR_IF_NOT(prefix->memory_resource == this);
117  FFL_CODDING_ERROR_IF_NOT(prefix->size > min_allocation_size);
118  size_t user_allocation_size = prefix->size - min_allocation_size;
119  FFL_CODDING_ERROR_IF_NOT(user_allocation_size == bytes);
120  FFL_CODDING_ERROR_IF_NOT(prefix->alignment == alignment);
121  suffix_t *suffix{ reinterpret_cast<suffix_t *>(reinterpret_cast<char *>(prefix) + prefix->size) - 1 };
122  FFL_CODDING_ERROR_IF_NOT(suffix->pattern == busy_block_suffix_pattern);
123  prefix->pattern = free_block_prefix_pattern;
124  suffix->pattern = free_block_suffix_pattern;
125 
126  _aligned_free(prefix);
127  }
128 
129  bool do_is_equal(memory_resource const & other) const noexcept override {
130  return (&other == static_cast<memory_resource const *>(this));
131  }

133  void increment_busy_block_count() {
134  size_t prev_busy_blocks_count = busy_blocks_count_.fetch_add(1, std::memory_order_relaxed);
135  FFL_CODDING_ERROR_IF_NOT(prev_busy_blocks_count >= 0);
136  }
137 
138  void decrement_busy_block_count() {
139  size_t prev_busy_blocks_count = busy_blocks_count_.fetch_sub(1, std::memory_order_relaxed);
140  FFL_CODDING_ERROR_IF_NOT(prev_busy_blocks_count > 0);
141  }
